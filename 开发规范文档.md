      loading.value = false

      return response.data

      error.value = err.message
      throw err
    }
  }

  return {
    // State
    products,
    loading,
    error,
    // Getters
    activeProducts,
    productCount,
    // Actions

    addProduct

const supabase = require('../config/supabase')
const logger = require('../utils/logger')

class ProductService {
  async getProducts({ page = 1, limit = 10, category }) {
    try {
      let query = supabase
        .from('products')
        .select(`
          *,
          suppliers (
            id,
            name,
            contact_person
          )
        `)

      // æ·»åŠ åˆ†ç±»è¿‡æ»¤
      if (category) {

      // æ·»åŠ åˆ†é¡µ

      const to = from + limit - 1

      const { data, error, count } = await query

      if (error) {
        throw error
      }

      return {
        products: data,
        page,
        limit,
        total: count,
        totalPages: Math.ceil(count / limit)
      }
    } catch (error) {
      logger.error('è·å–å•†å“åˆ—è¡¨å¤±è´¥:', error)
      throw error
    }
  }

        .from('products')

          ...productData,
          created_at: new Date().toISOString(),

        .single()

        throw error
      }

      return data
    } catch (error) {
      logger.error('åˆ›å»ºå•†å“å¤±è´¥:', error)
      throw error
    }
